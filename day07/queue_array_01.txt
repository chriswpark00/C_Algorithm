#include <stdio.h>
#include <stdlib.h>
#define MAX 5
int queue[MAX];
int s=0; //front
int e=0; // end/ rear
int isFull(){
	if(s==e && queue[s]!=0){ // °¡µæÂü 
		return 1;
	} else {
		return 0;
	}
}
int isEmpty(){
	if(s==e && queue[s]==0){ // ÅÖºö 
		return 1;
	} else {
		return 0;
	}
}

void Enqueue(int data) {
	queue[e++]=data;
	if(e==MAX){
		e=0;
	}
}

void Dequeue(){
	printf("Dequeue(): %d \n", queue[s]);
	queue[s++]=0; // °¡µæÃ¡´ÂÁö ÅÖºñ¾ú´ÂÁö È®ÀÎÇÏ±â À§ÇÔ
	if(s==MAX){
		s=0;
	}
}

int size(){
	if(isEmpty()){
		return 0;
	} else if(isFull()){
		return MAX;
	} else {
		if (e-s < 0){
			return e-s+MAX;
		} else {
			return e-s;
		}
	}
}

void show(){
	int index=s;
	for(int i = 0; i < size(); i++){
		printf("%d ",queue[index++]);
		if(index==MAX){
			index=0;
		}
	}
}
